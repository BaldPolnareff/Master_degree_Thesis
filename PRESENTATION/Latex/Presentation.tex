  \section{Caratterizzazione della polvere}

    \begin{frame}{Requisiti della polvere}
        \onslide*<1>{
        \begin{center}
          Quali sono i \textbf{requisiti} fondamentali per una polvere SLS ready?
        \end{center}  
        }

        \begin{itemize}
          \item <2-> Granulometria
          \item <3-> Morfologia
          \item <4> Stabilità termica
          
          
        \end{itemize}

        \onslide*<2>{La polvere deve avere una distribuzione media tra 20 e 80 $\mu m$}
        \onslide*<3>{Le particelle devono essere sferiche, non vuote, per consentire la formazione efficace di colli di sinterizzazione}
        \onslide*<4>{La polvere deve rimanere termicamente inalterata durante la stampa}

      \end{frame}
